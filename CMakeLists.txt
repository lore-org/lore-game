# Setup

cmake_minimum_required(VERSION 4.0)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# MacOS-specific declarations

if (NOT APPLE)
    add_compile_definitions(PARALLEL=true)
else()
    set(CMAKE_OSX_ARCHITECTURES arm64)
endif()

# Setup vcpkg

execute_process(COMMAND git submodule update --init --recursive)
set(CMAKE_TOOLCHAIN_FILE "${CMAKE_SOURCE_DIR}/vcpkg/scripts/buildsystems/vcpkg.cmake" CACHE STRING "Vcpkg toolchain file")

# Create project

project(lore-game VERSION 1.0.0)

# Generate config.hpp from config.json

file(READ ${CMAKE_SOURCE_DIR}/config.json CONFIG_FILE)
file(GENERATE
    OUTPUT ${CMAKE_SOURCE_DIR}/src/engine/config.hpp
    CONTENT
"// Autogenerated from config.json
#pragma once
#include <nlohmann/json.hpp>
inline const nlohmann::json config = nlohmann::json::parse(R\"(${CONFIG_FILE})\");"
    TARGET ${PROJECT_NAME}
)

# Add all .cpp files as sources

file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS src/*.cpp)

# Set entrypoint for project

add_executable(${PROJECT_NAME} src/main.cpp)

# Adding non-vcpkg packages

include(FetchContent)

FetchContent_Declare(
    raylib
    GIT_REPOSITORY https://github.com/raysan5/raylib.git
    GIT_TAG master
)
FetchContent_Declare(
    discord-rpc
    GIT_REPOSITORY https://github.com/EclipseMenu/discord-presence.git
    GIT_TAG main
)

FetchContent_MakeAvailable(
    discord-rpc
    raylib
)

# Adding vcpkg packages

find_package(CURL REQUIRED)
find_package(nlohmann_json CONFIG REQUIRED)
find_package(fmt CONFIG REQUIRED)

# Add libraries to project

target_link_libraries(${PROJECT_NAME} PUBLIC
    discord-rpc
    raylib

    CURL::libcurl
    nlohmann_json::nlohmann_json
    fmt
)

# Copy /resources

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${PROJECT_SOURCE_DIR}/resources
    ${PROJECT_BINARY_DIR}/resources
    COMMENT "Copying /resources to Build directory"
)
